FONKSİYON compute_disorder(t_stack_check *stack):
    // 1. GÜVENLİK KONTROLÜ
    EĞER stack mevcut değilse VEYA stack->size < 2 ise:
        Ondalıklı 0.0 değerini döndür.

    // Ondalıklı (double) hassas sayaçlarımızı başlatıyoruz
    Hatalar_Sayaci = 0.0
    Toplam_Ciftler  = 0.0

    Gezici_I = stack->head // Taramaya en tepedeki elemandan başlıyoruz

    // 2. İÇ İÇE ÇİFT TARAMA DÖNGÜSÜ
    DÖNGÜ: Gezici_I mevcut olduğu sürece:
        
        // Gezici_J'yi her zaman Gezici_I'nin bir altındaki elemandan başlatıyoruz
        Gezici_J = Gezici_I->next
        
        DÖNGÜ: Gezici_J mevcut olduğu sürece:
            Toplam_Ciftler = Toplam_Ciftler + 1.0
            
            // Eğer üstteki elemanın değeri, alttakinden BÜYÜKSE bu bir sıralama hatasıdır!
            EĞER Gezici_I->value > Gezici_J->value ise:
                Hatalar_Sayaci = Hatalar_Sayaci + 1.0
                
            Gezici_J = Gezici_J->next // Alttaki geziciyi bir aşağı kaydır
            
        Gezici_I = Gezici_I->next // Üstteki geziciyi bir aşağı kaydır

    // 3. MATEMATİKSEL ORAN HESABI
    EĞER Toplam_Ciftler == 0.0 ise:
        Ondalıklı 0.0 döndür.
        
    // Hataların toplam çiftlere oranını ondalıklı olarak döndür (Örn: 0.45)
    DÖNDÜR (Hatalar_Sayaci / Toplam_Ciftler)
